program test;
variables 
    x, i : int;
    y, resultado : float;

void function escribir ();
{
    write("Hola mundo!");
}

void function dibujar ();
{
    from i = 0 to 10 do {
        line(i * 10, vertical);
        arc(i * 10, 180);
    }
}

float function sumar (int x, float y);
{
    escribir()
    return (x + y);
}

principal() {
    x = 20;
    y = 5;

    resultado = (sumar(x, y) + 5) * 2;
    write(resultado);
    dibujar()
}
