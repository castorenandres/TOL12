program FibonacciRecursive;
variables
    fibo : int;

int function fibonacci (int n);
{
    if (n < 0) then {
        write("No existe");
        return(-1);
    } else {
        if (n == 0) then {
            return (0);
        } else {
            if (n == 1 | n == 2) then {
                return (1);
            } else {
                return (fibonacci(n - 1) + fibonacci(n - 2));
            }
        }
    }
}

principal() {
    fibo = fibonacci(10);
    write(fibo);
}
